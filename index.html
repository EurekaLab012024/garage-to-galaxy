window.onload = function () { const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); // Set canvas size canvas.width = window.innerWidth; canvas.height = window.innerHeight; // Draw a test circle ctx.fillStyle = 'orange'; ctx.beginPath(); ctx.arc(canvas.width / 2, canvas.height / 2, 50, 0, Math.PI * 2); ctx.fill(); // Draw some text ctx.fillStyle = 'white'; ctx.font = '24px Arial'; ctx.fillText('Garage to Galaxy Test Render', 50, 50); };
